    **1. Create efb-212/Views/Components/CurrencyBadge.swift:**
    A reusable SwiftUI view that displays a traffic-light currency indicator.

    ```swift
    struct CurrencyBadge: View {
        let status: CurrencyStatus
        let label: String

        var body: some View {
            HStack(spacing: 6) {
                Circle()
                    .fill(statusColor)
                    .frame(width: 12, height: 12)
                Text(label)
                    .font(.subheadline)
            }
        }

        private var statusColor: Color {
            switch status {
            case .current: return .green
            case .warning: return .yellow
            case .expired: return .red
            }
        }
    }
    ```
