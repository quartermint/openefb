    ```swift
    struct SearchBar: View {
        @Binding var text: String
        let placeholder: String
        let onSubmit: () -> Void

        var body: some View {
            HStack(spacing: 8) {
                Image(systemName: "magnifyingglass")
                    .foregroundColor(.secondary)
                    .accessibilityHidden(true)  // decorative per UI-SPEC accessibility

                TextField(placeholder, text: $text)
                    .textFieldStyle(.plain)
                    .font(.subheadline)
                    .autocorrectionDisabled()
                    .textInputAutocapitalization(.characters)  // ICAO IDs are uppercase
                    .onSubmit(onSubmit)

                if !text.isEmpty {
                    Button(action: { text = "" }) {
                        Image(systemName: "xmark.circle.fill")
                            .foregroundColor(.secondary)
                    }
                    .accessibilityLabel("Clear search")
                }
            }
            .padding(8)
            .background(.regularMaterial)
            .clipShape(RoundedRectangle(cornerRadius: 8))
        }
    }
    ```

    **SearchViewModel.swift** — `@Observable @MainActor final class SearchViewModel`:

    Properties:
    - `let databaseService: any DatabaseServiceProtocol`
    - `var query: String = ""`
    - `var results: [Airport] = []`
    - `var isSearching: Bool = false`
    - `private var searchTask: Task<Void, Never>?`

    Methods:
    - `func search()`:
      1. Cancel previous `searchTask`
      2. If `query.count < 2`, clear results, return
      3. Create new Task with 300ms debounce (`try await Task.sleep(nanoseconds: 300_000_000)`)
      4. Set `isSearching = true`
      5. Call `databaseService.searchAirports(query: query, limit: 20)`
      6. Set `results`
      7. Set `isSearching = false`

    - `func clear()`:
      `query = ""`
      `results = []`

    Per UI-SPEC: Empty search shows no results, no error toast. Error state for search: "Silent failure -- search results clear" per copywriting contract.

    **MapContainerView.swift** — The MapContainerView must compose ALL map-tab components per the UI-SPEC layout contract.

    **Note:** `SearchResultsList` referenced below is a **private struct defined within MapContainerView.swift** (not a separate file). It is a simple VStack/List rendering search results with airport ICAO, name, and city, styled with `.regularMaterial` background and 300pt max width. It does NOT need its own file — it is an implementation detail of MapContainerView.

    ```swift
    // Private helper struct defined within MapContainerView.swift
    private struct SearchResultsList: View {
        let results: [Airport]
        let onTap: (String) -> Void

        var body: some View {
            VStack(alignment: .leading, spacing: 0) {
                ForEach(results) { airport in
                    Button {
                        onTap(airport.icao)
                    } label: {
                        VStack(alignment: .leading, spacing: 2) {
                            Text(airport.icao)
                                .font(.subheadline.bold())
                            Text(airport.name)
                                .font(.caption)
                                .foregroundStyle(.secondary)
                        }
                        .padding(.horizontal, 12)
                        .padding(.vertical, 8)
                    }
                    .buttonStyle(.plain)
                    if airport.id != results.last?.id {
                        Divider()
                    }
                }
            }
            .background(.regularMaterial)
            .clipShape(RoundedRectangle(cornerRadius: 8))
            .frame(maxWidth: 300)
        }
    }

    struct MapContainerView: View {
        @Environment(AppState.self) private var appState

        // Create or receive services and view models
        // These should be created here or passed via environment/init

        var body: some View {
            ZStack {
                // Layer 1: Full-screen map
                MapView(mapService: mapService, mapViewModel: mapViewModel)
                    .ignoresSafeArea(edges: .top)

                // Layer 2: Floating UI overlays
                VStack(spacing: 0) {
                    // Top bar
                    HStack {
                        // Top-left: Search bar + NEAREST button
                        VStack(alignment: .leading, spacing: 8) {
                            // Search bar (collapsible — shows when activated)
                            if showingSearch {
                                SearchBar(text: $searchViewModel.query, placeholder: "Search airports (ICAO, name, city)", onSubmit: { searchViewModel.search() })
                                    .frame(maxWidth: 300)

                                // Search results dropdown (private struct within this file)
                                if !searchViewModel.results.isEmpty {
                                    SearchResultsList(results: searchViewModel.results, onTap: { icao in
                                        mapViewModel.onAirportTapped(icao: icao)
                                        showingSearch = false
                                        searchViewModel.clear()
                                    })
                                }
                            }
                        }
                        .padding(.leading, 16)

                        Spacer()

                        // Top-right: Map controls stack
                        VStack(spacing: 8) {
                            MapControlsView(mapViewModel: mapViewModel)
                        }
                        .padding(.trailing, 16)
                    }
                    .padding(.top, 60) // below status bar

                    Spacer()

                    // Bottom stack
                    VStack(spacing: 8) {
                        // TFR disclaimer banner (conditional — per UI-SPEC)
                        if appState.visibleLayers.contains(.tfrs) {
                            Text("TFR DATA IS SAMPLE ONLY — NOT FOR NAVIGATION")
                                .font(.caption2)
                                .fontWeight(.semibold)
                                .foregroundColor(.white)
                                .frame(maxWidth: .infinity)
                                .padding(.vertical, 6)
                                .background(Color.red)
                        }

                        // "No VFR charts downloaded" banner (conditional — per UI-SPEC)
                        // Show when no MBTiles files exist in Application Support/Charts/

                        // Nearest airport HUD (positioned above instrument strip)
                        HStack {
                            Spacer()
                            NearestAirportHUD(
                                nearestAirport: nearestAirportVM.nearestAirport,
                                distance: nearestAirportVM.nearestDistance,
                                bearing: nearestAirportVM.nearestBearing
                            )
                        }
                        .padding(.horizontal, 16)

                        // Instrument strip (bottom bar)
                        InstrumentStripView()
                            .padding(.horizontal, 16)
                            .padding(.bottom, 8)
                    }
                }
            }
            // Airport info sheet
            .sheet(isPresented: Binding(get: { appState.isPresentingAirportInfo }, set: { appState.isPresentingAirportInfo = $0 })) {
                AirportInfoSheet(airportICAO: appState.selectedAirportID ?? "")
            }
            // Nearest airport list sheet
            .sheet(isPresented: Binding(get: { appState.isPresentingNearestList }, set: { appState.isPresentingNearestList = $0 })) {
                NearestAirportView(
                    entries: nearestAirportVM.nearestAirports,
                    onDirectTo: { nearestAirportVM.setDirectTo(airport: $0) },
                    onAirportTap: { mapViewModel.onAirportTapped(icao: $0) }
                )
            }
            .task {
                // Start location service
                await locationService.startTracking()
                // Start reachability monitoring
                reachabilityService.start()
            }
            .onChange(of: appState.ownshipPosition) { _, newPosition in
                guard let position = newPosition else { return }
                // Update nearest airport (throttled inside VM)
                nearestAirportVM.updateNearest(from: position)
                // Update ownship on map
                mapService.updateOwnship(location: position, heading: appState.track)

                // First GPS fix animation
                if !appState.firstLocationReceived {
                    appState.firstLocationReceived = true
                    mapService.animateToLocation(position.coordinate, zoom: 10.0)
                }
            }
        }
    }
    ```
