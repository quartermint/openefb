    ```swift
    struct InstrumentStripView: View {
        @Environment(AppState.self) private var appState

        var body: some View {
            HStack(spacing: 0) {
                InstrumentCell(label: "GS", value: gsValue, unit: "KT")
                Divider().frame(height: 30)
                InstrumentCell(label: "ALT", value: altValue, unit: "FT")
                Divider().frame(height: 30)
                InstrumentCell(label: "VS", value: vsValue, unit: "FPM")
                Divider().frame(height: 30)
                InstrumentCell(label: "TRK", value: trkValue, unit: "deg")

                if appState.activeFlightPlan || appState.directToAirport != nil {
                    Divider().frame(height: 30)
                    InstrumentCell(label: "DTG", value: dtgValue, unit: "NM")
                    Divider().frame(height: 30)
                    InstrumentCell(label: "ETE", value: eteValue, unit: "")
                }
            }
            .padding(.horizontal, 16)
            .padding(.vertical, 8)
            .background(.ultraThinMaterial)
            .clipShape(RoundedRectangle(cornerRadius: 16))
            .overlay {
                if !appState.gpsAvailable {
                    // Subtle "No GPS" indicator overlaid
                    Text("No GPS")
                        .font(.caption2)
                        .foregroundColor(.secondary)
                        .padding(4)
                        .background(.ultraThinMaterial)
                        .clipShape(Capsule())
                        .transition(.opacity.animation(.easeIn(duration: 0.3)))
                }
            }
        }

        private var gsValue: String {
            guard appState.gpsAvailable else { return "---" }
            return "\(Int(appState.groundSpeed))"
        }
        private var altValue: String {
            guard appState.gpsAvailable else { return "---" }
            return "\(Int((appState.altitude / 10).rounded() * 10))"  // round to nearest 10
        }
        private var vsValue: String {
            guard appState.gpsAvailable else { return "---" }
            let vs = Int(appState.verticalSpeed)
            return vs >= 0 ? "+\(vs)" : "\(vs)"
        }
        private var trkValue: String {
            guard appState.gpsAvailable else { return "---" }
            return String(format: "%03.0f", appState.track)  // 3-digit with leading zeros
        }
        private var dtgValue: String {
            guard let dtg = appState.distanceToNext else { return "---" }
            return String(format: "%.1f", dtg)
        }
        private var eteValue: String {
            guard let ete = appState.estimatedTimeEnroute else { return "---" }
            let minutes = Int(ete / 60)
            let hours = minutes / 60
            let mins = minutes % 60
            if hours > 0 { return "\(hours)h\(String(format: "%02d", mins))m" }
            return "\(mins)m"
        }
    }

    // Nested InstrumentCell per UI-SPEC component inventory:
    // .caption2 label, .title3.monospaced.semibold value, .caption2 unit, 60pt min width, 4pt horizontal padding
    struct InstrumentCell: View {
        let label: String
        let value: String
        let unit: String

        var body: some View {
            VStack(spacing: 4) {
                Text(label)
                    .font(.caption2)
                    .fontWeight(.medium)
                    .foregroundColor(.secondary)
                HStack(alignment: .lastTextBaseline, spacing: 2) {
                    Text(value)
                        .font(.system(.title3, design: .monospaced))
                        .fontWeight(.semibold)
                    if !unit.isEmpty {
                        Text(unit)
                            .font(.caption2)
                            .foregroundColor(.secondary)
                    }
                }
            }
            .frame(minWidth: 60)
            .padding(.horizontal, 4)
            .accessibilityElement(children: .combine)
            .accessibilityLabel("\(label) \(value) \(unit)")
        }
    }
    ```

    Do NOT animate instrument strip value changes per UI-SPEC animation rules: "numeric data must update instantly for pilot situational awareness."

      2. Query `databaseService.nearestAirports(to: position.coordinate, count: 10)`
      3. For each airport, compute distance and bearing using `CLLocation` extensions
      4. Sort by distance
      5. Set `nearestAirport` to closest, `nearestDistance`, `nearestBearing`
      6. Set `nearestAirports` to full sorted list
      7. Update `lastQueryPosition`

      Set `appState.directToAirport = airport`, `appState.activeFlightPlan = true`
      Compute `appState.distanceToNext` from current ownship to airport
      Compute `appState.estimatedTimeEnroute` from distance and current ground speed

      `results = []`
